  <script>
    // Manage in-memory Video list
    let videos = [
      {
        id: 1,
        title: 'Odisha CHSE Marksheet Guide',
        youtube: 'https://www.youtube.com/watch?v=abc123',
        website: 'https://chseodisha.nic.in'
      }
    ];

    const form = document.getElementById('addVideoForm');
    const tableBody = document.querySelector('#videoTable tbody');

    function renderVideos() {
      tableBody.innerHTML = '';
      for (const v of videos) {
        const tr = document.createElement('tr');
        
        // ID
        const tdId = document.createElement('td');
        tdId.textContent = v.id;
        tr.appendChild(tdId);

        // Title
        const tdTitle = document.createElement('td');
        tdTitle.textContent = v.title;
        tr.appendChild(tdTitle);

        // YouTube Link
        const tdYT = document.createElement('td');
        const ytLink = document.createElement('a');
        ytLink.href = v.youtube;
        ytLink.target = '_blank';
        ytLink.rel = 'noopener noreferrer';
        ytLink.textContent = 'Open';
        ytLink.style.color = '#fa9847';
        tdYT.appendChild(ytLink);
        tr.appendChild(tdYT);

        // Website Link
        const tdWeb = document.createElement('td');
        if (v.website?.trim()) {
          const webLink = document.createElement('a');
          webLink.href = v.website;
          webLink.target = '_blank';
          webLink.rel = 'noopener noreferrer';
          webLink.textContent = 'Visit';
          webLink.style.color = '#cb6c21';
          tdWeb.appendChild(webLink);
        } else {
          tdWeb.textContent = '‚Äî';
          tdWeb.style.color = '#888';
        }
        tr.appendChild(tdWeb);

        // Actions
        const tdActions = document.createElement('td');
        tdActions.classList.add('actions');
        const delBtn = document.createElement('button');
        delBtn.className = 'btn-delete';
        delBtn.title = 'Delete';
        delBtn.innerHTML = 'üóëÔ∏è';
        delBtn.onclick = () => {
          if (confirm(`Delete video "${v.title}"?`)) {
            videos = videos.filter(video => video.id !== v.id);
            renderVideos();
          }
        };
        tdActions.appendChild(delBtn);
        tr.appendChild(tdActions);

        tableBody.appendChild(tr);
      }
    }

    form.addEventListener('submit', e => {
      e.preventDefault();
      const title = form.title.value.trim();
      const youtubeLink = form.youtubeLink.value.trim();
      const websiteLink = form.websiteLink.value.trim();

      if (!title || !youtubeLink) {
        alert('Please enter Title and YouTube video link.');
        return;
      }
      const newId = videos.length > 0 ? videos[videos.length - 1].id + 1 : 1;
      videos.push({
        id: newId,
        title,
        youtube: youtubeLink,
        website: websiteLink
      });

      renderVideos();

      form.reset();
    });

    // Initial render
    renderVideos();
  </script>
