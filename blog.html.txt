  <script>
    // Manage in-memory Blog list
    let blogs = [
      {
        id: 1,
        title: 'Odisha CHSE Result ‚Äî Step by Step Download',
        banner: '', // empty means no pic
        description: 'Learn how to quickly download your Odisha 12th marksheet from CHSE Odisha.',
        link: 'https://chseodisha.nic.in'
      }
    ];
    // Will hold banner as data-url for new blogs
    let bannerDataURL = '';

    const form = document.getElementById('addBlogForm');
    const blogListDiv = document.getElementById('blogList');
    const bannerInput = document.getElementById('banner');
    const bannerPreview = document.getElementById('bannerPreview');

    // Show preview and keep image as base64-url for display
    bannerInput.addEventListener('change', function() {
      const file = this.files[0];
      if (!file) {
        bannerPreview.style.display = 'none';
        bannerDataURL = '';
        return;
      }
      const reader = new FileReader();
      reader.onload = function(e) {
        bannerPreview.src = e.target.result;
        bannerPreview.style.display = 'block';
        bannerDataURL = e.target.result;
      };
      reader.readAsDataURL(file);
    });

    function renderBlogs() {
      blogListDiv.innerHTML = '';
      for (const b of blogs) {
        const card = document.createElement('div');
        card.className = 'blog-card';

        // Banner
        if (b.banner) {
          const bannerImg = document.createElement('img');
          bannerImg.src = b.banner;
          bannerImg.alt = "Banner";
          bannerImg.className = "blog-banner";
          card.appendChild(bannerImg);
        } else {
          const placeholder = document.createElement('div');
          placeholder.className = 'blog-banner';
          placeholder.style.display = 'flex';
          placeholder.style.alignItems = 'center';
          placeholder.style.justifyContent = 'center';
          placeholder.style.color = '#fdaf58';
          placeholder.style.background = '#fff6e0';
          placeholder.textContent = '‚Äî';
          card.appendChild(placeholder);
        }

        // Details
        const details = document.createElement('div');
        details.className = 'blog-details';
        const title = document.createElement('div');
        title.className = 'blog-title';
        title.textContent = b.title;
        details.appendChild(title);

        const desc = document.createElement('div');
        desc.className = 'blog-desc';
        desc.textContent = b.description;
        details.appendChild(desc);

        if (b.link && b.link.trim()) {
          const linkDiv = document.createElement('div');
          linkDiv.className = 'blog-link';
          const a = document.createElement('a');
          a.href = b.link; a.target = "_blank"; a.rel="noopener noreferrer"; a.textContent = b.link;
          linkDiv.appendChild(a);
          details.appendChild(linkDiv);
        }

        card.appendChild(details);

        // Delete button
        const delBtn = document.createElement('button');
        delBtn.className = 'btn-delete';
        delBtn.title = 'Delete';
        delBtn.innerHTML = 'üóëÔ∏è';
        delBtn.onclick = () => {
          if (confirm(`Delete blog "${b.title}"?`)) {
            blogs = blogs.filter(blog => blog.id !== b.id);
            renderBlogs();
          }
        };
        card.appendChild(delBtn);

        blogListDiv.appendChild(card);
      }
    }
    // Handle Form submission
    form.addEventListener('submit', e => {
      e.preventDefault();
      const title = form.title.value.trim();
      const description = form.description.value.trim();
      const link = form.link.value.trim();

      const newId = blogs.length > 0 ? blogs[blogs.length - 1].id + 1 : 1;

      blogs.push({
        id: newId,
        title,
        banner: bannerDataURL,
        description,
        link
      });

      renderBlogs();

      // Reset form and banner preview
      form.reset();
      bannerPreview.style.display = 'none';
      bannerPreview.src = '';
      bannerDataURL = '';
    });

    // Initial render
    renderBlogs();
  </script>
